428  inline int fast_round(double value)
429  {
430 #if ((defined _MSC_VER && defined _M_X64) || (defined __GNUC__ && defined __x86_64__ && defined __SSE2__ && !defined __APPLE__)) && !defined(__CUDACC__)
431  __m128d t = _mm_set_sd(value);
432  return _mm_cvtsd_si32(t);
433 #elif defined _MSC_VER && defined _M_IX86
434  int t;
435  __asm
436  {
437  fld value;
438  fistp t;
439  }
440  return t;
441 #elif defined _MSC_VER && defined _M_ARM && defined HAVE_TEGRA_OPTIMIZATION
442  TEGRA_ROUND(value);
443 #elif defined CV_ICC || defined __GNUC__
444 # ifdef HAVE_TEGRA_OPTIMIZATION
445  TEGRA_ROUND(value);
446 # else
447  return (int)lrint(value);
448 # endif
449 #else
450  double intpart, fractpart;
451  fractpart = modf(value, &intpart);
452  if ((fabs(fractpart) != 0.5) || ((((int)intpart) % 2) != 0))
453  return (int)(value + (value >= 0 ? 0.5 : -0.5));
454  else
455  return (int)intpart;
456 #endif
457  }
